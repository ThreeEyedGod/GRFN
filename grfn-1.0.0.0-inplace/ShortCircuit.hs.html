    4 -- | Module for nifty functions leveraging lazy evaluation
    5 module ShortCircuit
    6   ( -- * Type classes
    7     HasFalse (..),
    8     HasTrue (..),
    9     Shortcircuit (..),
   10 
   11     -- * Short circuit tests
   12     isFalse,
   13     if',
   14     unless',
   15     (??),
   16     (||),
   17     (&&),
   18     firstTrueOf,
   19     lastFalseOf,
   20     firstFalseOf,
   21 
   22     -- * Monadic short circuits
   23     orM,
   24     andM,
   25     firstTrueOfM,
   26     lastFalseOfM,
   27     firstFalseOfM,
   28   )
   29 where
   30 
   31 import Control.Monad
   32 import Data.Maybe
   33 import Prelude hiding ((&&), (||))
   34 
   35 -- | Types with a defined false value.
   36 class HasFalse a where
   37   -- | A false value for this type.
   38   -- If 'Shortcircuit' @a@ holds, @isTrue false == False@ must hold.
   39   false :: a
   40 
   41 -- | Types with a defined true value.
   42 class HasTrue a where
   43   -- | A true value for this type.
   44   -- If 'Shortcircuit' @a@ holds, @isTrue true == True@ must hold.
   45   true :: a
   46 
   47 -- | Types that support short circuits.
   48 class Shortcircuit a where
   49   -- | Whether the value is true-like (i.e. not false-like).
   50   isTrue :: a -> Bool
   51 
   52 -- | Whether the value is false-like (i.e. not true-like).
   53 isFalse :: (Shortcircuit a) => a -> Bool
   54 isFalse = not . isTrue
   55 
   56 -- | @if then else@ generalised to 'Shortcircuit'.
   57 if' :: (Shortcircuit a) => a -> b -> b -> b
   58 if' x a b
   59   | isTrue x = a
   60   | otherwise = b
   61 
   62 -- | The opposite of 'if''.
   63 unless' :: (Shortcircuit a) => a -> b -> b -> b
   64 unless' x a b
   65   | isFalse x = a
   66   | otherwise = b
   67 
   68 infix 2 ??, ||, &&
   69 
   70 -- | Like 'if'', but with different argument order, allowing infix use.
   71 (??) :: (Shortcircuit a) => b -> b -> a -> b
   72 a ?? b = \x -> if' x a b
   73 
   74 -- | 'Prelude.||' generalised to 'Shortcircuit'.
   75 (||) :: (Shortcircuit a) => a -> a -> a
   76 (||) = join if'
   77 
   78 -- | 'Prelude.&&' generalised to 'Shortcircuit'.
   79 (&&) :: (Shortcircuit a) => a -> a -> a
   80 (&&) = join unless'
   81 
   82 -- | Returns the first true-ish value from a list, or 'false'.
   83 firstTrueOf :: (Shortcircuit a, HasFalse a) => [a] -> a
   84 firstTrueOf = foldr (||) false
   85 
   86 -- | Returns the last false-ish value from a list, or 'true'.
   87 lastFalseOf :: (Shortcircuit a, HasTrue a) => [a] -> a
   88 lastFalseOf = foldr (&&) true
   89 
   90 -- | Returns the first False-sh value of a provided list
   91 firstFalseOf :: (Shortcircuit a, HasTrue a) => [a] -> a
   92 firstFalseOf = foldr (flip (&&)) true -- foldl (&&) true stan reported that foldl was space-leaking hence moved to the current one
   93 
   94 -- | Short-circuit two actions, performing the second only if the first returned a false-ish value.
   95 orM :: (Monad m, Shortcircuit a) => m a -> m a -> m a
   96 orM a b = a >>= \x -> (pure x ?? b) x
   97 
   98 -- | Short-circuit two actions, performing the second only if the first returned a true-ish value.
   99 andM :: (Monad m, Shortcircuit a) => m a -> m a -> m a
  100 andM a b = a >>= \x -> (b ?? pure x) x
  101 
  102 -- | Short-circuit a list of actions, performing only until a true-ish value is found, or the list exhausted.
  103 firstTrueOfM :: (Monad m, Shortcircuit a, HasFalse a) => [m a] -> m a
  104 firstTrueOfM = foldr orM (pure false)
  105 
  106 -- | Short-circuit a list of actions, performing only until a false-ish value is found, or the list exhausted.
  107 lastFalseOfM :: (Monad m, Shortcircuit a, HasTrue a) => [m a] -> m a
  108 lastFalseOfM = foldr andM (pure true)
  109 
  110 -- | Returns the first False-ish value from a monadic list
  111 firstFalseOfM :: (Monad m, Shortcircuit a, HasTrue a) => [m a] -> m a
  112 firstFalseOfM = foldr (flip andM) (pure true) -- foldl andM (pure true) stan reported that foldl was space - leaking hence moved to the current one
  113 
  114 instance HasTrue Bool where
  115   true = True
  116 
  117 instance HasFalse Bool where
  118   false = False
  119 
  120 instance Shortcircuit Bool where
  121   isTrue = id
  122 
  123 instance HasFalse (Maybe a) where
  124   false = Nothing
  125 
  126 instance Shortcircuit (Maybe a) where
  127   isTrue = isJust
  128 
  129 instance Shortcircuit (Either a b) where
  130   isTrue (Left _) = False
  131   isTrue (Right _) = True
