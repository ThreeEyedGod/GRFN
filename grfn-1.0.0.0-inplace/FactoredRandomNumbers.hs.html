    2 {--
    3    The Adam Kalai Algorithm implmemented in this module (see Readme for more details)
    4               Input: Integer n > 0.
    5 
    6               Output: A uniformly random number 1 ≤ r ≤ n.
    7 
    8                   1. Generate a sequence n ≥ s1 ≥ s2 ≥ ··· ≥ sl = 1 by choosing
    9                   s1 ∈ {1, 2,..., n} and si+1 ∈ {1, 2,...,si}, until reaching 1.
   10                   2. Let r be the product of the prime si’s.
   11                   3. If r ≤ n, output r with probability r/n.
   12                   4. Otherwise, RESTART.
   13 --}

   90 -- | Spin up t threads of function f in parallel and return what's executed first
   91 spinUpThreads :: IO a -> Int -> IO (Maybe a)
   92 spinUpThreads f t = withPool t $ \pool -> parallelFirst pool $ replicate t (Just <$> f)

  102 -- | Convert async.race from Either-Or to Maybe
  103 raceJust :: IO a -> IO (Maybe a)
  104 raceJust a =  do 
  105           r <- race a a
  106           case r of 
  107               Left u -> pure $ Just u 
  108               Right v -> pure $ Just v

  114 -- | Compute cores to actually use when called. 
  115 coresToUse :: IO (Int,Int)
  116 coresToUse = do
  117   nCores <- getNumProcessors
  118   nNumCapabilities <- getNumCapabilities
  119   let nEfficiencyCores = 4
  120   setNumCapabilities $ max ((nCores-nEfficiencyCores) * 4) nNumCapabilities
  121   nNumCapabilitiesSet <- getNumCapabilities
  122   pure (nCores, nNumCapabilitiesSet)

  140 -- | This is the Entry Function with a Integer bound
  141 -- Provide an integer input and it should generate a tuple of a number less than the input integer and its prime factors
  142 genARandomPreFactoredNumberLTEn :: Integer -> IO (Either Text (Integer, [Integer]))
  143 genARandomPreFactoredNumberLTEn x | x <= 0 = pure $ Left $ pack "Invalid"
  144 genARandomPreFactoredNumberLTEn 1 = pure $ Right (1, [1])
  145 genARandomPreFactoredNumberLTEn n = do
  146   candidateTuple <- potentialResult n
  147   if' ((fst `is` (< n)) candidateTuple) (pure $ Right candidateTuple) (genARandomPreFactoredNumberLTEn n) -- else keep doing till success

  149 -- | Provided an Integer List, throws up a candidate Int and its prime factors for further assessment
  150 filterPrimesProduct :: [Integer] -> (Integer, [Integer])
  151 --filterPrimesProduct xs = result where result@(_, sq) = (product sq, filter isPrimeOr1 xs) -- note: product [] = 1
  152 filterPrimesProduct xs = result where result@(_, sq) = (product sq, onlyPrimes xs) -- note: product [] = 1

  154 parFilter :: (NFData a) => Int -> (a -> Bool) -> [a] -> [a]
  155 parFilter stratParm p = S.withStrategy (S.parListSplitAt stratParm S.rpar S.rpar) . filter p

  157 -- | parallel reduction of a composite list of integers into primefactors 
  158 onlyPrimes :: [Integer] -> [Integer]
  159 onlyPrimes xs = parFilter (length xs `div` 2) isPrimeOr1 xs

  161 -- | Provided an Integer, throws up a candidate Int and its factors for further assessment
  162 potentialResult :: Integer -> IO (Integer, [Integer])
  163 potentialResult n = makeList n <&> filterPrimesProduct

  165 -- | Provided an Integer, creates a sequence of random integers LTE n in decreasing order,
  166 -- possibly with multiples ending at a single 1
  167 makeList :: Integer -> IO [Integer]
  168 makeList 1 = pure []
  169 makeList n | n > 1 = (getRndMInt >=>: makeList) (1, n)
  170 makeList _ = error "Out of bound Arg"

  172 -- | Get a Random Integer with uniform probability in the range [1,n]
  173 getRndMInt :: (Integer, Integer) -> IO Integer
  174 getRndMInt (l, u) | l <= u && l > 0 = max l . min u <$> uniformRM (l, u) globalStdGen
  175 getRndMInt _ = error "Malformed Range"

  183 -- | True if input is prime or 1
  184 -- Primality testing is one key to peformance of this algo
  185 -- the isOdd and greater than 3 is for the use of bailliePSW primality 
  186 -- using bailliePSW in place of the standard isPrime leads to 75% reduction in time !!!
  187 isPrimeOr1 :: Integer -> Bool
  188 isPrimeOr1 n | n > 0 = (n == 1 || n == 3) || (n > 3 && isOdd n && bailliePSW n) --bailliePSW requires that n > 3 and that input is Odd
  189 isPrimeOr1 _ = error "Invalid Arg "
