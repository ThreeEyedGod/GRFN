    3    The Adam Kalai Algorithm implmemented in this module
    4               Input: Integer n > 0.
    5 
    6               Output: A uniformly random number 1 ≤ r ≤ n.
    7 
    8                   1. Generate a sequence n ≥ s1 ≥ s2 ≥ ··· ≥ sl = 1 by choosing
    9                   s1 ∈ {1, 2,..., n} and si+1 ∈ {1, 2,...,si}, until reaching 1.
   10                   2. Let r be the product of the prime si’s.
   11                   3. If r ≤ n, output r with probability r/n.
   12                   4. Otherwise, RESTART.
   13 --}

   63 -- | Takes an Int for Bitsize value to operate on range [2 ^ y, 2 ^ y + 1 - 1].  This function leverages parallel execution
   64 -- Provide an integer input and it should generate a tuple of a number in the range [2^y, 2^y+1 -1] and its prime factors
   65 -- In the event that the parallel call fails and return Nothing, a recovery through a non-parallel call is attempted.
   66 preFactoredNumOfBitSizePar :: Int -> IO (Either Text (Int, [Int]))
   67 preFactoredNumOfBitSizePar 1 = pure $ Right (1, [1])
   68 preFactoredNumOfBitSizePar n | n > 1 = fromMaybe <$> preFactoredNumOfBitSize n <*> preFactoredNumOfBitSizeParMaybe n
   69 preFactoredNumOfBitSizePar _ = pure $ Left $ pack "Invalid"

   71 -- | Failable Parallel preFactored Number given BitSize
   72 -- Provide an integer input and it should generate a tuple of a number in the range [2^y, 2^y+1 -1] and its prime factors
   73 preFactoredNumOfBitSizeParMaybe :: Int -> IO (Maybe (Either Text (Int, [Int])))
   74 preFactoredNumOfBitSizeParMaybe 1 = pure $ Just $ Right (1, [1])

   78 -- | Spin up t threads of function f in parallel and return what's executed first
   79 spinUpThreads :: IO a -> Int -> IO (Maybe a)
   80 spinUpThreads f t = withPool t $ \pool -> parallelFirst pool $ replicate t (Just <$> f)

   97 -- | This is the Entry Function with a Int bound
   98 -- Provide an integer input and it should generate a tuple of a number less than the input integer and its prime factors
   99 genARandomPreFactoredNumberLTEn :: Int -> IO (Either Text (Int, [Int]))
  100 genARandomPreFactoredNumberLTEn x | x <= 0 = pure $ Left $ pack "Invalid"
  101 genARandomPreFactoredNumberLTEn 1 = pure $ Right (1, [1])
  102 genARandomPreFactoredNumberLTEn n = do
  103   candidateTuple <- potentialResult n
  104   if' ((fst `is` (< n)) candidateTuple) (pure $ Right candidateTuple) (genARandomPreFactoredNumberLTEn n) -- keep doing till success

  106 -- | Provided an Int List, throws up a candidate Int and its prime factors for further assessment
  107 filterPrimesProduct :: [Int] -> (Int, [Int])
  108 filterPrimesProduct xs = result where result@(ps, sq) = (product sq, filter isPrimeOr1 xs) -- note: product [] = 1
  109 
  110 -- | Provided an Int, throws up a candidate Int and its factors for further assessment
  111 potentialResult :: Int -> IO (Int, [Int])
  112 potentialResult n = makeList n <&> filterPrimesProduct

  114 -- | Provided an Int, creates a sequence of random integers LTE n in decreasing order,
  115 -- possibly with multiples ending at a single 1
  116 makeList :: Int -> IO [Int]
  117 makeList 1 = pure []
  118 makeList n | n > 1 = (getRndMInt >=>: makeList) (1, n)
  119 makeList _ = error "Out of bound Arg"

  121 -- | Get a Random Integer with uniform probability in the range [1,n]
  122 getRndMInt :: (Int, Int) -> IO Int
  123 getRndMInt (l, u) | l <= u && l > 0 = max l . min u <$> uniformRM (l, u) globalStdGen
  124 getRndMInt _ = error "Malformed Range"
