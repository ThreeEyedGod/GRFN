    2 {--
    3    The Adam Kalai Algorithm implmemented in this module (see Readme for more details)
    4               Input: Integer n > 0.
    5 
    6               Output: A uniformly random number 1 ≤ r ≤ n.
    7 
    8                   1. Generate a sequence n ≥ s1 ≥ s2 ≥ ··· ≥ sl = 1 by choosing
    9                   s1 ∈ {1, 2,..., n} and si+1 ∈ {1, 2,...,si}, until reaching 1.
   10                   2. Let r be the product of the prime si’s.
   11                   3. If r ≤ n, output r with probability r/n.
   12                   4. Otherwise, RESTART.
   13 --}

   99 -- | Spin up t threads of function f in parallel and return what's executed first
  100 spinUpThreads :: IO a -> Int -> IO (Maybe a)
  101 spinUpThreads f t = withPool t $ \pool -> parallelFirst pool $ replicate t (Just <$> f)

  105 spinUpActions :: IO a -> Int -> IO (Maybe a)

  111 -- | Convert async.race from Either-Or to Maybe
  112 raceJust :: IO a -> IO (Maybe a)
  113 raceJust a = do
  114   r <- race a a
  115   case r of
  116     Left u -> pure $ Just u
  117     Right v -> pure $ Just v

  123 -- | Compute cores to actually use when called.
  124 coresToUse :: IO (Int, Int)
  125 coresToUse = do
  126   nCores <- getNumProcessors
  127   nNumCapabilities <- getNumCapabilities
  128   let nEfficiencyCores = 4
  129   setNumCapabilities $ max ((nCores - nEfficiencyCores) * 4) nNumCapabilities
  130   nNumCapabilitiesSet <- getNumCapabilities
  131   pure (nCores, nNumCapabilitiesSet)

  148 -- | This is the Entry Function with a Integer bound
  149 -- Provide an integer input and it should generate a tuple of a number less than the input integer and its prime factors
  150 genARandomPreFactoredNumberLTEn :: Integer -> IO (Either Text (Integer, [Integer]))
  151 genARandomPreFactoredNumberLTEn x | x <= 0 = pure $ Left $ pack "Invalid"
  152 genARandomPreFactoredNumberLTEn 1 = pure $ Right (1, [1])
  153 genARandomPreFactoredNumberLTEn n = do
  154   candidateTuple <- potentialResult n
  155   if' ((fst `is` (< n)) candidateTuple) (pure $ Right candidateTuple) (genARandomPreFactoredNumberLTEn n) -- else keep doing till success

  157 -- | Provided an Integer List, throws up a candidate Int and its prime factors for further assessment
  158 filterPrimesProduct :: [Integer] -> (Integer, [Integer])
  159 filterPrimesProduct xs = result where result@(_, sq) = (product sq, onlyPrimesFrom xs) -- note: product [] = 1

  161 parFilter :: (NFData a) => Strats -> Int -> (a -> Bool) -> [a] -> [a]
  162 parFilter strat stratParm p = case strat of
  163   Chunk -> S.withStrategy (S.parListChunk stratParm S.rdeepseq) . filter p
  164   Buffer -> S.withStrategy (S.parBuffer stratParm S.rdeepseq) . filter p
  165   Split -> S.withStrategy (S.parListSplitAt stratParm S.rpar S.rpar) . filter p
  166 
  167 -- | parallel reduction of a composite list of integers into primefactors
  168 -- select the strategy based on the size range to be built in
  169 onlyPrimesFrom :: [Integer] -> [Integer]
  170 onlyPrimesFrom xs
  171   | head xs < 10 ^ 9 = filter isPrimeOr1 xs
  172   | head xs < 2 ^ 32 = parFilter Buffer (length xs `div` 2) isPrimeOr1 xs
  173   | head xs < 2 ^ 62 = parFilter Chunk (length xs `div` 3) isPrimeOr1 xs
  174   | otherwise = parFilter Split (length xs `div` 3) isPrimeOr1 xs

  179 -- | Provided an Integer, throws up a candidate Int and its factors for further assessment
  180 potentialResult :: Integer -> IO (Integer, [Integer])
  181 potentialResult n = mkList n <&> filterPrimesProduct

  183 -- | Provided an Integer, creates a sequence of random integers LTE n in decreasing order,
  184 -- possibly with multiples ending at a single 1
  185 mkList :: Integer -> IO [Integer]
  186 mkList 1 = pure []
  187 mkList n | n > 1 = (getRndMInt >=>: mkList) (1, n)
  188 mkList _ = error "Out of bound Arg"

  190 -- | Get a Random Integer with uniform probability in the range [1,n]
  191 getRndMInt :: (Integer, Integer) -> IO Integer
  192 getRndMInt (l, u) | l <= u && l > 0 = max l . min u <$> uniformRM (l, u) globalStdGen
  193 getRndMInt _ = error "Malformed Range"

  201 -- | True if input is prime or 1
  202 -- Primality testing is one key to peformance of this algo
  203 -- the isOdd and greater than 3 is for the use of bailliePSW primality
  204 -- using bailliePSW in place of the standard isPrime leads to 75% reduction in time !!!
  205 isPrimeOr1 :: Integer -> Bool
  206 isPrimeOr1 n | n > 0 = (n == 1 || n == 3) || (n > 3 && isOdd n && bailliePSW n) -- bailliePSW requires that n > 3 and that input is Odd
  207 isPrimeOr1 _ = error "Invalid Arg "
