    2    The Adam Kalai Algorithm implmemented in this module
    3               Input: Integer n > 0.
    4 
    5               Output: A uniformly random number 1 ≤ r ≤ n.
    6 
    7                   1. Generate a sequence n ≥ s1 ≥ s2 ≥ ··· ≥ sl = 1 by choosing
    8                   s1 ∈ {1, 2,..., n} and si+1 ∈ {1, 2,...,si}, until reaching 1.
    9                   2. Let r be the product of the prime si’s.
   10                   3. If r ≤ n, output r with probability r/n.
   11                   4. Otherwise, RESTART.

   61 -- | Takes an Int for Bitsize value to operate on range [2 ^ y, 2 ^ y + 1 - 1].  This function leverages parallel execution
   62 -- Provide an integer input and it should generate a tuple of a number in the range [2^y, 2^y+1 -1] and its prime factors
   63 -- In the event that the parallel call fails and return Nothing, a recovery through a non-parallel call is attempted.
   64 preFactoredNumOfBitSizePar :: Int -> IO (Either Text (Int, [Int]))
   65 preFactoredNumOfBitSizePar 1 = pure $ Right (1, [1])
   66 preFactoredNumOfBitSizePar n | n > 1 = fromMaybe <$> preFactoredNumOfBitSize n <*> preFactoredNumOfBitSizeParMaybe n
   67 preFactoredNumOfBitSizePar _ = pure $ Left $ pack "Invalid"

   70 -- | Failable Parallel preFactored Number given BitSize

   72 preFactoredNumOfBitSizeParMaybe :: Int -> IO (Maybe (Either Text (Int, [Int])))
   73 preFactoredNumOfBitSizeParMaybe 1 = pure $ Just $ Right (1, [1])

   78 -- | Spin up t threads of function f in parallel and return what's executed first
   79 spinUpThreads :: IO a -> Int -> IO (Maybe a)
   80 spinUpThreads f t = withPool t $ \pool -> parallelFirst pool $ replicate t (Just <$> f)

   82 -- | Takes an Int as a Bitsize value to operate on range [2 ^ y, 2 ^ y + 1 - 1]
   83 -- Provide an integer input and it should generate a tuple of a number in the range [2^y, 2^y+1 -1] and its prime factors
   84 preFactoredNumOfBitSize :: Int -> IO (Either Text (Int, [Int]))
   85 preFactoredNumOfBitSize n | n <= 0 = pure $ Left $ pack "Invalid"
   86 preFactoredNumOfBitSize 1 = pure $ Right (1, [1])
   87 preFactoredNumOfBitSize n | n > 1 = iterateWhile ((2 ^ n) <|) (genARandomPreFactoredNumberLTEn (2 ^ (n + 1) - 1))
   88 preFactoredNumOfBitSize _ = pure $ Left $ pack "Invalid"

   92 -- | An operator to compare the Right first value of the (Int, [Int]) to an Int for lesser-than predicate
   93 (<|) :: Int -> Either Text (Int, [Int]) -> Bool
   94 bound <| eOR = case eOR of
   95   Left _ -> False
   96   Right v -> fst v < bound

   98 -- | This is the Entry Function with a Int bound
   99 -- Provide an integer input and it should generate a tuple of a number less than the input integer and its prime factors
  100 genARandomPreFactoredNumberLTEn :: Int -> IO (Either Text (Int, [Int]))
  101 genARandomPreFactoredNumberLTEn x | x <= 0 = pure $ Left $ pack "Invalid"
  102 genARandomPreFactoredNumberLTEn 1 = pure $ Right (1, [1])
  103 genARandomPreFactoredNumberLTEn n = potentialResult n >>= go n
  104   where
  105     go n' candidateTuple
  106       | fst candidateTuple <= n' = pure $ Right candidateTuple
  107       | otherwise = genARandomPreFactoredNumberLTEn n' -- keep doing till result occurs

  109 -- | Provided an Int List, throws up a candidate Int and its factors for further assessment
  110 filterPrimesProduct :: [Int] -> (Int, [Int])
  111 filterPrimesProduct xs = result where result@(ps, sq) = (product sq, filter isPrimeOr1 xs) -- note: product [] = 1
  112 
  113 -- | Provided an Int, throws up a candidate Int and its factors for further assessment
  114 potentialResult :: Int -> IO (Int, [Int])
  115 potentialResult n = makeList n <&> filterPrimesProduct

  117 -- | Provided an Int, creates a sequence of random integers LTE n in decreasing order,
  118 -- possibly with multiples ending at a single 1
  119 makeList :: Int -> IO [Int]
  120 makeList 1 = pure [] -- pure [] also works
  121 makeList n | n > 1 = (getRndMInt >=>: makeList) (1, n)
  122 makeList _ = error "Out of bound Arg"

  124 -- | Get a Random Integer with uniform probability in the range [1,n]
  125 getRndMInt :: (Int, Int) -> IO Int
  126 getRndMInt (l, u) | l <= u && l > 0 = max l . min u <$> uniformRM (l, u) globalStdGen
  127 getRndMInt _ = error "Malformed Range"
